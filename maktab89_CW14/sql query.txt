select c.customer_id, r.customer_id, r.rental_date, r.return_date
from customer c right join rental r
on c.customer_id=r.customer_id
where r.customer_id is null;

select i.film_id, f.film_id, COUNT(f.film_id), f.title
from rental r inner join inventory i
on r.inventory_id=i.inventory_id
inner join film f on i.film_id=f.film_id
group by (i.film_id, f.film_id)
order by count(f.film_id) desc;

select a.first_name, a.actor_id, fa.film_id, f.length
from actor a inner join film_actor fa
on a.actor_id=fa.actor_id
inner join film as f
on fa.film_id=f.film_id
where f.length>60;

2 new=============
select a.actor_id, concat(a.first_name, ' ', a.last_name) full_name, count(fa.film_id)
from actor a inner join film_actor fa
on a.actor_id=fa.actor_id
group by a.actor_id
order by count(fa.film_id) desc;
